f_nb <- function(idx, loo_type, classifier_type) {
  if (loo_type == "lgocv") {
    pvec <- grp != grp[idx]
  } else {
    pvec <- seq(length(y)) != idx
  }
  if (classifier_type == "nb") {
    predict(e1071::naiveBayes(X[pvec,], y[pvec]), X[idx,])  
  }
}

plot.comparison <- function(metric, thresh) {
  cmf <- merge(cmf0[, c("class", metric)], 
             cmf1[, c("class", metric)], 
             by=c("class"),
             suffixes=c("_without_interphase", "_with_interphase"))

  with_interphase_label <- paste(metric, "_with_interphase", sep="" )
  without_interphase_label <- paste(metric, "_without_interphase", sep="" )
  cmf$with_interphase <- cmf[, with_interphase_label ]
  cmf$without_interphase <- cmf[, without_interphase_label ]
  
  p <- 
  ggplot(cmf, aes(without_interphase, with_interphase)) + 
    geom_text(data=subset(cmf, abs(without_interphase - with_interphase) > thresh ), 
              aes(label=class)) + 
    geom_point() + 
    geom_abline() + coord_equal() + theme_bw() + 
    xlab(without_interphase_label) + 
    ylab(with_interphase_label) + ggtitle(sprintf("Classifier = %s", 
                                                  classifier_type))
    
  ggsave(sprintf("%s_comparison.pdf", metric), p, width=8, height=8)
}

create.groups <- function(cpd.leave.out) {
  # leave out the specified compound
  D0 <- droplevels(subset(D, Compound != cpd.leave.out))
  
  # create partition
  D0 <- ddply(D0, .(Compound), function(X) {X$idx <- sample(NROW(X)) %% 4; X} )
  D_tr <- subset(D0, idx %in% c(0,1,2) )
  D_va <- subset(D0, idx == 3)
  
  dname <- file.path("groups", cpd.leave.out)

  write.class <- function(P, tag)  {
    write.table(P$MOAi, 
                file.path(dname, paste("y", tag, ".txt", sep="")), 
                quote=F, row.names=F, col.names=F)
  }
  
  write.feats <- function(P, tag)  {
    
    write.table(P[,feat.names], 
                file.path(dname, paste("x", tag, ".txt", sep="")), 
                quote=F, row.names=F, col.names=F)
  }
  dir.create(dname, recursive=T)
  write.class(D_tr, "tr")
  write.class(D_va, "va")
  write.feats(D_tr, "tr")
  write.feats(D_va, "va")
  file.copy('../results/master/0f2a248a_mLMNN2.5/xall.txt', dname)
}
